Go to: Firebase Console → Firestore Database → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Users - allow all authenticated reads
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated() && request.auth.uid == userId;
    }
    
    // Conversations - simpler rule for testing
    match /conversations/{conversationId} {
      // First try: allow if authenticated and user is in participants
      // For list queries, this checks each document in the result
      allow read: if isAuthenticated() && 
        request.auth.uid in resource.data.participants;
      
      allow create: if isAuthenticated() && 
        request.auth.uid in request.resource.data.participants;
      
      allow update, delete: if isAuthenticated() && 
        request.auth.uid in resource.data.participants;
      
      // Messages
      match /messages/{messageId} {
        allow read, write: if isAuthenticated();
        allow create: if isAuthenticated() && 
          request.resource.data.senderId == request.auth.uid;
      }
    }
    
    // Listings
    match /listings/{listingId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated() && resource.data.hostId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.hostId == request.auth.uid;
    }
    
    // Bookings
    match /bookings/{bookingId} {
      allow read: if isAuthenticated() && (
        resource.data.guestId == request.auth.uid ||
        resource.data.hostId == request.auth.uid
      );
      allow create: if isAuthenticated() && request.resource.data.guestId == request.auth.uid;
      allow update: if isAuthenticated() && (
        resource.data.guestId == request.auth.uid ||
        resource.data.hostId == request.auth.uid
      );
    }
    
    // Onboarding drafts
    match /onboardingDrafts/{draftId} {
      allow read, write: if isAuthenticated() && resource.data.userId == request.auth.uid;
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
    }
  }
}

NOTE: The key difference - using "allow read" instead of separate "allow list" and "allow get"
This works for both queries and single document reads.
